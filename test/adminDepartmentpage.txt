if (content) {
  // DELETE OLD VIDEO
  if (body.videoPath && body.videoPath !== content.videoPath) {
    deleteFile(content.videoPath);
  }

  // DELETE OLD BANNERS
  if (Array.isArray(content.bannerImages)) {
    content.bannerImages.forEach(oldImg => {
      if (!body.bannerImages?.includes(oldImg)) {
        deleteFile(oldImg);
      }
    });
  }

  // DELETE OLD SERVICE IMAGES
  if (Array.isArray(content.services)) {
    content.services.forEach((oldService, i) => {
      const newService = body.services?.[i];

      if (oldService.imageUrl && oldService.imageUrl !== newService?.imageUrl) {
        deleteFile(oldService.imageUrl);
      }

      const oldDetail = oldService.detailedContent || {};
      const newDetail = newService?.detailedContent || {};

      if (oldDetail.image1Path !== newDetail.image1Path) {
        deleteFile(oldDetail.image1Path);
      }

      if (oldDetail.image2Path !== newDetail.image2Path) {
        deleteFile(oldDetail.image2Path);
      }
    });
  }
}
